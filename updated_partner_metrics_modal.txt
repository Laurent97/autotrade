                        storeVisits: {

                        }
                      }
                    }));
                  }}
                />

                <label className="block text-sm font-medium text-gray-700 dark:text-gray-300 mb-1">This Week (Auto-calculated)</label>

                  readOnly

                    setPartnerMetrics(prev => ({
                      ...prev,
                      [selectedUser.id]: {
                        ...prev[selectedUser.id],
                        commissionRate: value / 100
                      }
                    }));
                  }}
                />

                    onChange={(e) => {
                      setPartnerMetrics(prev => ({
                        ...prev,
                        [selectedUser.id]: {
                          ...prev[selectedUser.id],
                          isVerified: e.target.checked
                        }
                      }));
                    }}
                  />
